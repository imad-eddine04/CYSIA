#!/bin/bash

# 1. Récupérer le fichier en paramètre ou en saisie
if [ -z "$1" ]; then
    read -p "Saisir le nom du fichier : " fich
else
    fich=$1
fi

# 2. Vérifier que le fichier existe
if [ ! -f "$fich" ]; then
    echo "Le fichier n'existe pas."
    exit
fi

# 3. Lire chaque ligne du fichier
while read login nom prenom
do
    # si le login existe dans /etc/passwd :
    if grep "^$login:" /etc/passwd > /dev/null; then
        echo "$login   $nom   $prenom"
    fi

done < "$fich"
